24.gender(){
    local no_of_arms=2
    local no_of_legs=2
    if [ $1 = 'male' ]
    then
        echo 'male'
        male $no_of_arms $no_of_legs
    elif [ $1 = 'female' ]
    then
        echo 'female'
        female $no_of_arms $no_of_legs
    else
        echo "error in input"
    fi


}
